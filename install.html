// -*- coding: utf-8 -*- // HYPHÆLIUM API SERVER v1.1 // © 2026 Hyphælium. All rights reserved. // // Six endpoints: // POST /register Create account manually // POST /login Subscriber logs in // POST /save Save a breadcrumb // GET /crumbs Fetch breadcrumbs for vault // GET /health Server status // POST /create-payment Create Stripe payment intent (embedded checkout) // POST /webhook Stripe webhook - auto-create accounts on payment 'use strict'; const express = require('express'); const cors = require('cors'); const bcrypt = require('bcryptjs'); const jwt = require('jsonwebtoken'); const { createClient } = require('@supabase/supabase-js'); const nodemailer = require('nodemailer'); const stripe = require('stripe'); require('dotenv').config(); const app = express(); // ── Supabase client (service role - server-side only) ── const supabase = createClient( process.env.SUPABASE_URL, process.env.SUPABASE_SERVICE_KEY ); // ── Stripe client ────────────────────────────────────── const stripeClient = stripe(process.env.STRIPE_SECRET_KEY); // ── Email transporter ────────────────────────────────── const mailer = nodemailer.createTransport({ host: process.env.EMAIL_HOST || 'smtp.gmail.com', port: parseInt(process.env.EMAIL_PORT || '587'), secure: false, auth: { user: process.env.EMAIL_USER, pass: process.env.EMAIL_PASS } }); // ── Plans ────────────────────────────────────────────── const PLANS = { personal: { name: 'Personal', price: 99, label: '99p/month' }, small_team: { name: 'Small Team', price: 499, label: '£4.99/month' }, team: { name: 'Team', price: 999, label: '£9.99/month' } }; function planFromAmount(pence) { if (pence >= 999) return 'team'; if (pence >= 499) return 'small_team'; return 'personal'; } // ── CORS ─────────────────────────────────────────────── const allowedOrigins = (process.env.ALLOWED_ORIGINS || '') .split(',').map(s => s.trim()).filter(Boolean); app.use(cors({ origin: (origin, callback) => { if (!origin) return callback(null, true); if (origin.startsWith('chrome-extension://')) return callback(null, true); if (allowedOrigins.some(o => origin.startsWith(o))) return callback(null, true); if (origin.includes('localhost') || origin.includes('127.0.0.1')) return callback(null, true); callback(new Error('Not allowed by CORS')); }, credentials: true })); // ── Body parsing ─────────────────────────────────────── app.use((req, res, next) => { if (req.path === '/webhook') { express.raw({ type: 'application/json' })(req, res, next); } else { express.json()(req, res, next); } }); // ── JWT middleware ───────────────────────────────────── function requireAuth(req, res, next) { const header = req.headers.authorization; if (!header || !header.startsWith('Bearer ')) { return res.status(401).json({ error: 'Not authenticated' }); } try { req.user = jwt.verify(header.slice(7), process.env.JWT_SECRET); next(); } catch(e) { return res.status(401).json({ error: e.name === 'TokenExpiredError' ? 'Session expired - please log in again' : 'Invalid token' }); } } // ── Rate limiting ────────────────────────────────────── const rateLimits = new Map(); function rateLimit(key, maxPerMinute) { const now = Date.now(); const recent = (rateLimits.get(key) || []).filter(t => now - t < 60000); if (recent.length >= maxPerMinute) return false; recent.push(now); rateLimits.set(key, recent); return true; } setInterval(() => { const cutoff = Date.now() - 60000; for (const [key, times] of rateLimits.entries()) { const fresh = times.filter(t => t > cutoff); if (!fresh.length) rateLimits.delete(key); else rateLimits.set(key, fresh); } }, 300000); // ════════════════════════════════════════════ // HEALTH CHECK // GET /health // ════════════════════════════════════════════ app.get('/health', async (req, res) => { try { const { data, error } = await supabase.from('hypha_users').select('id').limit(1); res.json({ status: 'ok', version: '1.1.0', database: error ? 'error' : 'connected', db_error: error ? error.message : null, db_code: error ? error.code : null, time: new Date().toISOString() }); } catch(e) { res.status(500).json({ status: 'error', message: e.message }); } }); // ════════════════════════════════════════════ // CREATE PAYMENT INTENT // POST /create-payment // Called by the subscribe page before card entry // Returns a client_secret for Stripe Elements // ════════════════════════════════════════════ app.post('/create-payment', async (req, res) => { const { plan, email } = req.body; if (!plan || !PLANS[plan]) { return res.status(400).json({ error: 'Invalid plan' }); } if (!email || !email.includes('@')) { return res.status(400).json({ error: 'Valid email required' }); } // Rate limit: 10 payment attempts per minute per IP if (!rateLimit(req.ip + ':payment', 10)) { return res.status(429).json({ error: 'Too many requests - please wait a moment' }); } try { const planInfo = PLANS[plan]; // Check if email already has active subscription const { data: existing } = await supabase .from('hypha_users') .select('id, active') .eq('email', email.toLowerCase()) .single(); if (existing?.active) { return res.status(409).json({ error: 'An active subscription already exists for this email address' }); } // Create or retrieve Stripe customer let customerId; const customers = await stripeClient.customers.list({ email, limit: 1 }); if (customers.data.length > 0) { customerId = customers.data[0].id; } else { const customer = await stripeClient.customers.create({ email }); customerId = customer.id; } // Create subscription with payment intent // Using setup_future_usage so card is saved for recurring billing const paymentIntent = await stripeClient.paymentIntents.create({ amount: planInfo.price, currency: 'gbp', customer: customerId, setup_future_usage: 'off_session', automatic_payment_methods: { enabled: true }, metadata: { plan, email, product: 'Hyphælium' }, description: `Hyphælium ${planInfo.name} - ${planInfo.label}` }); res.json({ clientSecret: paymentIntent.client_secret, customerId, plan, planName: planInfo.name, planLabel: planInfo.label, amount: planInfo.price }); } catch(e) { console.error('[create-payment]', e.message); res.status(500).json({ error: 'Could not create payment: ' + e.message }); } }); // ════════════════════════════════════════════ // CONFIRM PAYMENT & CREATE ACCOUNT // POST /confirm-payment // Called after Stripe payment succeeds // Creates account and sends welcome email // ════════════════════════════════════════════ app.post('/confirm-payment', async (req, res) => { const { paymentIntentId, email, password, plan } = req.body; if (!paymentIntentId || !email || !password || !plan) { return res.status(400).json({ error: 'Missing required fields' }); } if (password.length < 8) { return res.status(400).json({ error: 'Password must be at least 8 characters' }); } try { // Verify payment succeeded with Stripe const paymentIntent = await stripeClient.paymentIntents.retrieve(paymentIntentId); if (paymentIntent.status !== 'succeeded') { return res.status(400).json({ error: 'Payment has not completed - please try again' }); } // Verify email matches payment intent if (paymentIntent.metadata.email !== email.toLowerCase()) { return res.status(400).json({ error: 'Email does not match payment' }); } const normalEmail = email.toLowerCase().trim(); // Check not already registered const { data: existing } = await supabase .from('hypha_users') .select('id') .eq('email', normalEmail) .single(); if (existing) { // Already exists - just log them in const { data: user } = await supabase .from('hypha_users') .select('id, email, plan, active') .eq('email', normalEmail) .single(); const token = jwt.sign( { userId: user.id, email: user.email, plan: user.plan }, process.env.JWT_SECRET, { expiresIn: '30d' } ); return res.json({ ok: true, token, email: user.email, plan: user.plan, existing: true }); } // Create account const passwordHash = await bcrypt.hash(password, 12); const { data: newUser, error: insertError } = await supabase .from('hypha_users') .insert({ email: normalEmail, password_hash: passwordHash, stripe_customer_id: paymentIntent.customer, plan, subscribed_at: new Date().toISOString(), active: true }) .select('id, email, plan') .single(); if (insertError) throw insertError; // Issue JWT const token = jwt.sign( { userId: newUser.id, email: newUser.email, plan: newUser.plan }, process.env.JWT_SECRET, { expiresIn: '30d' } ); // Send welcome email await sendWelcomeEmail(normalEmail, plan); res.status(201).json({ ok: true, token, email: newUser.email, plan: newUser.plan }); } catch(e) { console.error('[confirm-payment]', e.message); res.status(500).json({ error: 'Account creation failed - please contact privacy@hyphaelium.com' }); } }); // ════════════════════════════════════════════ // LOGIN // POST /login // ════════════════════════════════════════════ app.post('/login', async (req, res) => { const { email, password } = req.body; if (!email || !password) { return res.status(400).json({ error: 'Email and password are required' }); } if (!rateLimit(req.ip + ':login', 10)) { return res.status(429).json({ error: 'Too many attempts - please wait a moment' }); } const normalEmail = email.toLowerCase().trim(); try { const { data: user, error } = await supabase .from('hypha_users') .select('id, email, password_hash, plan, active') .eq('email', normalEmail) .single(); if (error || !user) { return res.status(401).json({ error: 'Email or password not recognised' }); } if (!user.active) { return res.status(401).json({ error: 'Subscription not active - please check your billing at hyphaelium.com' }); } const valid = await bcrypt.compare(password, user.password_hash); if (!valid) { return res.status(401).json({ error: 'Email or password not recognised' }); } const token = jwt.sign( { userId: user.id, email: user.email, plan: user.plan }, process.env.JWT_SECRET, { expiresIn: '30d' } ); res.json({ ok: true, token, email: user.email, plan: user.plan }); } catch(e) { console.error('[login]', e.message); res.status(500).json({ error: 'Login failed - please try again' }); } }); // ════════════════════════════════════════════ // SAVE BREADCRUMB // POST /save // ════════════════════════════════════════════ app.post('/save', requireAuth, async (req, res) => { const { platform, route_path, identity_tag, topic, depth, boundary_count, boundary_flag } = req.body; if (!platform || !route_path) { return res.status(400).json({ error: 'Platform and route_path are required' }); } if (!rateLimit(req.user.userId + ':save', 60)) { return res.status(429).json({ error: 'Saving too quickly - please slow down' }); } try { const { error } = await supabase .from('hyphaelium_captures') .insert({ user_id: req.user.userId, captured_at: new Date().toISOString(), platform, route_path, identity_tag: identity_tag || 'untitled', topic: topic || 'general', depth: depth || 0, boundary_count: boundary_count || 0, boundary_flag: boundary_flag || false }); if (error) throw error; res.json({ ok: true }); } catch(e) { console.error('[save]', e.message); res.status(500).json({ error: 'Could not save breadcrumb' }); } }); // ════════════════════════════════════════════ // FETCH BREADCRUMBS // GET /crumbs // ════════════════════════════════════════════ app.get('/crumbs', requireAuth, async (req, res) => { const limit = Math.min(parseInt(req.query.limit || '500'), 2000); const platform = req.query.platform || null; const topic = req.query.topic || null; const search = req.query.search || null; if (!rateLimit(req.user.userId + ':crumbs', 30)) { return res.status(429).json({ error: 'Too many requests' }); } try { let query = supabase .from('hyphaelium_captures') .select('id, captured_at, platform, route_path, identity_tag, topic, depth, boundary_count, boundary_flag') .eq('user_id', req.user.userId) .order('captured_at', { ascending: false }) .limit(limit); if (platform) query = query.eq('platform', platform); if (topic) query = query.eq('topic', topic); if (search) query = query.ilike('identity_tag', '%' + search + '%'); const { data, error } = await query; if (error) throw error; res.json({ ok: true, crumbs: data || [], count: (data || []).length }); } catch(e) { console.error('[crumbs]', e.message); res.status(500).json({ error: 'Could not fetch breadcrumbs' }); } }); // ════════════════════════════════════════════ // STRIPE WEBHOOK // POST /webhook // ════════════════════════════════════════════ app.post('/webhook', async (req, res) => { const sig = req.headers['stripe-signature']; let event; try { event = stripeClient.webhooks.constructEvent( req.body, sig, process.env.STRIPE_WEBHOOK_SECRET ); } catch(e) { console.error('[webhook] Signature failed:', e.message); return res.status(400).json({ error: 'Invalid signature' }); } if (event.type === 'customer.subscription.deleted') { const custId = event.data.object.customer; if (custId) { await supabase .from('hypha_users') .update({ active: false }) .eq('stripe_customer_id', custId); console.log('[webhook] Subscription cancelled for customer:', custId); } } if (event.type === 'payment_intent.payment_failed') { const pi = event.data.object; console.log('[webhook] Payment failed:', pi.metadata?.email); } res.json({ received: true }); }); // ════════════════════════════════════════════ // EMAIL HELPERS // ════════════════════════════════════════════ async function sendWelcomeEmail(email, plan) { const planInfo = PLANS[plan] || PLANS.personal; Hyphælium carries. We carry. `); }); module.exports = app;
